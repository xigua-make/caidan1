                    radios.forEach(r => {
                        const label = r.closest('label');
                        if (r.checked) {
                            label.style.borderColor = 'var(--primary)';
                            label.style.backgroundColor = '#EEF2FF';
                        } else {
                            label.style.borderColor = 'var(--border)';
                            label.style.backgroundColor = 'transparent';
                        }
                    });
                });
                // 触发初始状态
                if (radio.checked) {
                    radio.dispatchEvent(new Event('change'));
                }
            });
        }

        // --- 2. 登录功能 ---
        function login() {
            const pwd = document.getElementById('passwordInput').value;
            if (!pwd) {
                showToast('请输入密码', 'error');
                return;
            }
            PASSWORD = pwd;
            document.getElementById('login-overlay').classList.add('hidden');
            document.getElementById('app-container').classList.remove('hidden');
            
            loadCodes();
            loadTodayRecords();
            showToast('登录成功', 'success');
        }

        document.getElementById('passwordInput').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') login();
        });

        // --- 3. 页面切换 ---
        function switchTab(tab) {
            const navItems = document.querySelectorAll('.nav-item');
            navItems.forEach((item, index) => {
                if (tab === 'dashboard' && index === 0) {
                    item.classList.add('active');
                } else if (tab === 'records' && index === 1) {
                    item.classList.add('active');
                } else if (index < 2) {
                    item.classList.remove('active');
                }
            });

            if (tab === 'dashboard') {
                document.getElementById('dashboard-page').classList.remove('hidden');
                document.getElementById('records-page').classList.add('hidden');
            } else if (tab === 'records') {
                document.getElementById('dashboard-page').classList.add('hidden');
                document.getElementById('records-page').classList.remove('hidden');
                loadTodayRecords();
            }
        }

        // --- 4. 加载数据 ---
        function loadCodes() {
            fetch('/api/licenses')
                .then(res => res.json())
                .then(data => {
                    codes = data.map(item => ({
                        code: item.code,
                        type: item.type || 'permanent',
                        max: item.max,
                        used: item.used,
                        permanent: item.permanent !== undefined ? item.permanent : (item.max === 0),
                        enabled: item.enabled,
                        createdAt: item.createdAt || new Date().toISOString()
                    }));
                    renderTable(codes);
                    updateStats();
                })
                .catch(err => {
                    console.error('加载失败:', err);
                    showToast('加载激活码数据失败', 'error');
                });
        }

        // 加载今日生成记录
        function loadTodayRecords() {
            fetch('/api/today-records')
                .then(res => res.json())
                .then(data => {
                    todayRecords = data || [];
                    renderRecordTable(todayRecords);
                    document.getElementById('todayCount').textContent = todayRecords.length;
                })
                .catch(err => {
                    console.error('加载记录失败:', err);
                    // 如果后端接口不存在，使用本地存储模拟
                    loadLocalRecords();
                });
        }

        // 本地存储模拟（如果后端接口不存在）
        function loadLocalRecords() {
            const today = new Date().toDateString();
            const stored = localStorage.getItem('todayRecords_' + today);
            todayRecords = stored ? JSON.parse(stored) : [];
            renderRecordTable(todayRecords);
            document.getElementById('todayCount').textContent = todayRecords.length;
        }

        // 刷新记录
        function refreshRecords() {
            loadTodayRecords();
            showToast('记录已刷新', 'success');
        }

        // --- 5. 渲染表格 ---
        function renderTable(data) {
            const tbody = document.getElementById('tableBody');
            const emptyState = document.getElementById('emptyState');
            tbody.innerHTML = '';

            if (data.length === 0) {
                emptyState.style.display = 'block';
                return;
            }
            emptyState.style.display = 'none';

            data.forEach(item => {
                const tr = document.createElement('tr');
                const typeInfo = CARD_TYPE_MAP[item.type] || CARD_TYPE_MAP['permanent'];
                
                tr.innerHTML = `
                    <td><span class="code-cell">${item.code}</span></td>
                    <td><span class="type-badge type-${item.type}">${typeInfo.name}</span></td>
                    <td>
                        <input type="number" 
                               id="m${item.code}" 
                               class="edit-input" 
                               value="${item.max}" 
                               placeholder="0=永久"
                               ${item.type === 'permanent' ? 'disabled style="background:#f3f4f6;cursor:not-allowed;"' : ''}
                        >
                    </td>
                    <td><input type="number" id="u${item.code}" class="edit-input" value="${item.used}"></td>
                    <td><input type="checkbox" id="e${item.code}" class="edit-checkbox" ${item.enabled ? 'checked' : ''}></td>
                    <td>
                        <button class="btn-save" onclick="save('${item.code}')">保存</button>
                        <button class="btn-icon" title="复制" onclick="copyCode('${item.code}')"><i class="ph ph-copy"></i></button>
                        <button class="btn-icon" title="删除" onclick="deleteCode('${item.code}')" style="color:var(--danger)"><i class="ph ph-trash"></i></button>
                    </td>
                `;
                tbody.appendChild(tr);
            });
        }

        // 渲染生成记录表格
        function renderRecordTable(data) {
            const tbody = document.getElementById('recordTableBody');
            const emptyState = document.getElementById('recordEmptyState');
            tbody.innerHTML = '';

            if (data.length === 0) {
                emptyState.style.display = 'block';
                document.getElementById('recordTable').style.display = 'none';
                return;
            }
            emptyState.style.display = 'none';
            document.getElementById('recordTable').style.display = 'table';

            data.forEach((item, index) => {
                const tr = document.createElement('tr');
                const typeInfo = CARD_TYPE_MAP[item.type] || CARD_TYPE_MAP['permanent'];
                
                tr.innerHTML = `
                    <td>${index + 1}</td>
                    <td><span class="code-cell">${item.code}</span></td>
                    <td><span class="type-badge type-${item.type}">${typeInfo.name}</span></td>
                    <td>${formatTime(item.createdAt)}</td>
                    <td>
                        <button class="btn-icon" title="复制" onclick="copyCode('${item.code}')"><i class="ph ph-copy"></i></button>
                    </td>
                `;
                tbody.appendChild(tr);
            });
        }

        // 格式化时间
        function formatTime(isoString) {
            const date = new Date(isoString);
            return date.toLocaleTimeString('zh-CN', {
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit'
            });
        }

        // --- 6. 统计与图表 ---
        function updateStats() {
            const total = codes.length;
            const active = codes.filter(c => c.enabled).length;
            const disabled = codes.filter(c => !c.enabled).length;
            const totalUsage = codes.reduce((sum, c) => sum + c.used, 0);

            document.getElementById('stat-total').innerText = total;
            document.getElementById('stat-active').innerText = active;
            document.getElementById('stat-disabled').innerText = disabled;
            document.getElementById('stat-usage').innerText = totalUsage;

            const day1 = codes.filter(c => c.type === 'day1').length;
            const day7 = codes.filter(c => c.type === 'day7').length;
            const perm = codes.filter(c => c.type === 'permanent').length;
            const exhausted = codes.filter(c => !c.permanent && c.max > 0 && c.used >= c.max).length;

            updateBar('bar-day1', 'val-day1', day1, total);
            updateBar('bar-day7', 'val-day7', day7, total);
            updateBar('bar-perm', 'val-perm', perm, total);
            updateBar('bar-exhausted', 'val-exhausted', exhausted, total);
        }

        function updateBar(barId, valId, val, total) {
            const percent = total === 0 ? 0 : (val / total) * 100;
            document.getElementById(barId).style.width = `${percent}%`;
            document.getElementById(valId).innerText = val;
        }

        // --- 7. 搜索过滤 ---
        function filterCodes() {
            const text = document.getElementById('searchInput').value.toLowerCase();
            const status = document.getElementById('statusFilter').value;
            const type = document.getElementById('typeFilter').value;
            
            const filtered = codes.filter(c => {
                const matchText = c.code.toLowerCase().includes(text);
                let matchStatus = true;
                let matchType = true;
                
                if (status === 'active') matchStatus = c.enabled;
                if (status === 'disabled') matchStatus = !c.enabled;
                if (type !== 'all') matchType = c.type === type;
                
                return matchText && matchStatus && matchType;
            });
            renderTable(filtered);
        }

        // --- 8. 保存与删除 ---
        function save(code) {
            const maxVal = document.getElementById(`m${code}`).value;
            const usedVal = document.getElementById(`u${code}`).value;
            const enabledVal = document.getElementById(`e${code}`).checked;

            const codeData = codes.find(c => c.code === code);
            
            fetch('/api/update', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    password: PASSWORD,
                    code: code,
                    max: Number(maxVal),
                    used: Number(usedVal),
                    permanent: codeData.type === 'permanent',
                    enabled: enabledVal
                })
            })
            .then(res => res.json())
            .then(data => {
                if (data.success) {
                    showToast('保存成功', 'success');
                    loadCodes();
                } else {
                    showToast('保存失败', 'error');
                }
            })
            .catch(err => {
                console.error(err);
                showToast('网络错误', 'error');
            });
        }

        function deleteCode(code) {
            if (!confirm("确定要删除这个激活码吗？此操作无法撤销。")) {
                return;
            }

            fetch('/api/delete', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    password: PASSWORD,
                    code: code
                })
            })
            .then(res => res.json())
            .then(data => {
                if (data.success) {
                    showToast('删除成功', 'success');
                    loadCodes();
                } else {
                    showToast('删除失败：' + (data.msg || '未知错误'), 'error');
                }
            })
            .catch(err => {
                console.error('删除错误:', err);
                showToast('网络错误', 'error');
            });
        }

        function copyCode(code) {
            navigator.clipboard.writeText(code).then(() => {
                showToast(`已复制: ${code}`, 'success');
            }).catch(() => {
                showToast('复制失败', 'error');
            });
        }

        // --- 9. 生成功能 ---
        function openModal() {
            document.getElementById('genModal').classList.add('open');
            // 重置单选框状态
            const radios = document.querySelectorAll('input[name="cardType"]');
            radios.forEach(r => {
                if (r.value === 'permanent') r.checked = true;
                r.dispatchEvent(new Event('change'));
            });
        }

        function closeModal() {
            document.getElementById('genModal').classList.remove('open');
            document.getElementById('genCount').value = 1;
        }

        function submitGeneration() {
            const count = Number(document.getElementById('genCount').value || 1);
            const cardType = document.querySelector('input[name="cardType"]:checked').value;
            const typeInfo = CARD_TYPE_MAP[cardType];

            fetch('/api/generate', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    password: PASSWORD,
                    count: count,
                    type: cardType,
                    permanent: typeInfo.permanent,
                    max: typeInfo.max
                })
            })
            .then(res => res.json())
            .then(data => {
                if (data.success) {
                    showToast(`成功生成 ${count} 个${typeInfo.name}`, 'success');
                    
                    // 添加到今日记录（模拟）
                    const newRecords = (data.codes || []).map(code => ({
                        code: code,
                        type: cardType,
                        createdAt: new Date().toISOString()
                    }));
                    todayRecords = [...todayRecords, ...newRecords];
                    
                    // 保存到本地存储（如果后端不支持）
                    const today = new Date().toDateString();
                    localStorage.setItem('todayRecords_' + today, JSON.stringify(todayRecords));
                    
                    closeModal();
                    loadCodes();
                    loadTodayRecords();
                } else {
                    showToast(data.msg || '生成失败', 'error');
                }
            })
            .catch(err => {
                console.error(err);
                showToast('网络错误', 'error');
            });
        }

        // --- 10. 导出功能 ---
        function exportData() {
            let csvContent = "data:text/csv;charset=utf-8,激活码,类型,最大次数,已用次数,是否启用\n";
            codes.forEach(c => {
                const typeInfo = CARD_TYPE_MAP[c.type] || CARD_TYPE_MAP['permanent'];
                csvContent += `${c.code},${typeInfo.name},${c.max},${c.used},${c.enabled ? '是' : '否'}\n`;
            });
            const encodedUri = encodeURI(csvContent);
            const link = document.createElement("a");
            link.setAttribute("href", encodedUri);
            link.setAttribute("download", "activation_codes.csv");
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            showToast('已导出 CSV', 'success');
        }

        function exportTodayRecords() {
            if (todayRecords.length === 0) {
                showToast('今日暂无生成记录', 'error');
                return;
            }

            const today = new Date().toLocaleDateString('zh-CN').replace(/\//g, '-');
            let csvContent = "data:text/csv;charset=utf-8,序号,激活码,类型,生成时间\n";
            
            todayRecords.forEach((record, index) => {
                const typeInfo = CARD_TYPE_MAP[record.type] || CARD_TYPE_MAP['permanent'];
                csvContent += `${index + 1},${record.code},${typeInfo.name},${formatTime(record.createdAt)}\n`;
            });
            
            const encodedUri = encodeURI(csvContent);
            const link = document.createElement("a");
            link.setAttribute("href", encodedUri);
            link.setAttribute("download", `今日生成记录_${today}.csv`);
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            showToast('已导出今日记录', 'success');
        }

                });

        }

        // --- Toast 提示 ---
        function showToast(message, type = 'success') {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            let icon = type === 'success' ? 'check-circle' : (type === 'error' ? 'warning-circle' : 'info');
            toast.innerHTML = `<i class="ph ph-${icon}" style="font-size: 1.2rem; color: ${type === 'success' ? 'var(--success)' : (type === 'error' ? 'var(--danger)' : 'var(--primary)')}"></i> <span>${message}</span>`;
            container.appendChild(toast);
            setTimeout(() => {
                toast.style.opacity = '0';
                toast.style.transform = 'translateX(100%)';
                toast.style.transition = 'all 0.3s';
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }
